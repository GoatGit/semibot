```python
import openpyxl

# 创建工作簿
wb = openpyxl.Workbook()

# 获取活跃工作表
ws = wb.active

# 添加表头
ws.append(["产品名", "数量", "金额"])

# 添加产品数据
ws.append(["产品A", 100, 5000])
ws.append(["产品B", 200, 8000])

# 保存成 xlsx 文件
wb.save("sales_report.xlsx")

print("文件已生成")
```
